

model("Another simple") {

	a : compartment("A")
	
	qf : property("Q!")
	r : property("R")
	
	module("So simple", version(2, 0, 0)) {

		par_group("The Q") {
			q : par_int("Q", [], 4)
		}
		
		var(a.qf, [], "Q!") {
			i := 0,
			r := 1,
			p:{
				i <- i+1,
				{ r <- r*i, iterate p }  if i <= q,
				r                        otherwise
			}
		}
	}
}
