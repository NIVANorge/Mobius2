

model("Another simple") {

	b : compartment("B")
	q : quantity("Q")
	
	i : index_set("I")
	distribute(b, i)
	d : connection("Grid", grid1d) { b* }
	
	
	sol : solver("Solver", "INCADascru", 0.1)
	solve(sol, b.q)

	module("So simple", version(1, 0, 0)) {
	
		test : property("Target") 
		
		var(b.test, []) { floor(5*0.5*(1 + sin((time.step=>[])*2*pi/150))) }
	
		var(b.q, []) @initial { 0 }
		
		
		flux(nowhere, b.q[d.specific], [day-1], "Flixflux")
			{ 0.01[day-1] }
			@specific { test }
	
	}
}
