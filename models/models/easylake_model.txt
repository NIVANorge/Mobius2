
model("EasyLake standalone") {
	
	load("../modules/easylake.txt", el : module("EasyLake"))
	
	sol : solver("EasyLake solver", "INCADascru", 0.1)
	
	epi : compartment("Epilimnion")
	hyp : compartment("Hypolimnion")
	lake : compartment("Lake")
	
	water : quantity("Water")
	heat  : quantity("Thermal energy")
	ice   : quantity("Ice")
	
	sol.solve(epi.water)
	sol.solve(hyp.water)
	sol.solve(lake.heat)
	sol.solve(epi.ice)
	
	lk : index_set("Lake")
	
	epi.distribute(lk)
	hyp.distribute(lk)
	lake.distribute(lk)
	
	
	#downstream : neighbor("Downstream", directed_tree) {
	#	*[epi river]
	#}
}
