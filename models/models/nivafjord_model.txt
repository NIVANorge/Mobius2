
model("NIVAFjord") {
	load("modules/nivafjord.txt", module("NIVAFjord"))
	load("modules/atmospheric.txt", module("Atmospheric"))
	
	layer_idx : index_set("Layer index")
	
	oair  : compartment("Ocean-air interface")
	layer : compartment("Fjord layer")
	sed   : compartment("Fjord sediment")
	bnd_layer : compartment("Boundary layer")
	
	water : quantity("Water")
	heat  : quantity("Thermal energy")
	h     : quantity("Sea level")
	
	distribute(layer, layer_idx)
	distribute(bnd_layer, layer_idx)
	
	#sol : solver("Simply solver", "INCADascru", 0.1)
	
	solve(solver("Simply solver", "INCADascru", 0.1), layer.water, sed.heat, bnd_layer.water)
}
